Accordion UI - JS


$(function() {
	// タイトルがクリックされたときの処理を定義する
	$('a.panel-heading').on('click', null, function(ev) {
		ev.preventDefault();

		/*
		クリックされた要素の
		hrefに指定されたidで
		対象の要素を取得して
		表示・非表示を切り替える
		*/
		$(this.hash).slideToggle();
	});
});

Tab UI - JS


$(function() {
	$('.tabs a').on('click', null, function(ev) {
		ev.preventDefault();

		/*
		タブメニューのアクティビティを切り替える
		*/
		// 一度全てのタブメニューから"active"を取り除く
		$('.tabs li').removeClass('active');
		// 該当のタブメニューにのみ"active"を付け加える
		$('.tabs li').has(this).addClass('active');

		/*
		タブパネルのアクティビティを切り替える
		*/
		// 一度全てのタブパネルから"active"を取り除く
		$('.tab-pane').removeClass('active');
		// 該当のタブパネルにのみ"active"を付け加える
		$(this.hash).addClass('active');
	});
});
